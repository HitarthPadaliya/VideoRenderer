float Border = BorderWidth / min(ObjectScale.x, ObjectScale.y);
float2 Center = float2(0.5, 0.5);

float Distance = distance(UV, Center);

float2 Delta = UV - Center;
float Theta = atan2(Delta.y, Delta.x);
float AngleNorm = frac((Theta / (2.0 * 3.14159)) + 0.5);

float DashMask = (fmod(AngleNorm * DashRepeat, 1.0) > DashGap) ? 1.0 : 0.0;

bool IsInRing = Distance < 0.5 && Distance > 0.5-Border;
float IsDashedRing = IsInRing * DashMask;

return lerp(BackgroundColor, BorderColor, IsDashedRing);
